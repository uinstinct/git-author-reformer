      use nucleo::{Config, Nucleo};
      use nucleo::pattern::{CaseMatching, Normalization};
      use std::sync::Arc;

      pub fn build_author_nucleo(items: &[AuthorIdentity]) -> Nucleo&lt;AuthorIdentity&gt; {
          let nucleo = Nucleo::new(Config::DEFAULT, Arc::new(|| {}), None, 1);
          let injector = nucleo.injector();
          for item in items {
              let display = format!("{} &lt;{}&gt;", item.name, item.email);
              injector.push(item.clone(), move |_, cols| { cols[0] = display.clone().into(); });
          }
          nucleo
      }

      pub fn apply_filter(nucleo: &mut Nucleo&lt;AuthorIdentity&gt;, query: &str) -> Vec&lt;AuthorIdentity&gt; {
          nucleo.pattern.reparse(0, query, CaseMatching::Ignore, Normalization::Smart, false);
          nucleo.tick(10);
          let snap = nucleo.snapshot();
          snap.matched_items(..).map(|m| m.data.clone()).collect()
      }
