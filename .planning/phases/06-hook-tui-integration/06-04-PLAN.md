       Screen::HookManageList { filter, matched, nucleo, selected, .. } => match key {
           Enter: NLL-clone email from matched[*selected]. Call remove_strip:
               Updated { .. } -> re-read via read_strip_list -> HookSuccess { state }
               HookDeleted    -> app.screen = Screen::HookSuccess { state: crate::hook::HookState::Absent }
               NotFound       -> app.screen = Screen::Err("email not found in strip list (unexpected)".into())
               Err(e)         -> app.screen = Screen::Err(e.to_string())

           Esc: app.screen = Screen::MainMenu { selected: 3 };
